  5  # A node can reference a "group", a profile-wide template to propagate multiple
  6  # TE keys to different YAML files at once
  7  group: "GRPSERIAL"
  8  # A tag is a label to classify jobs when displaying results
  9  tag: ["a", "b"]
 10  # Build rules => how to build the benchmark
 11  build:
 12    # Files used to build this node
 13    # it can be a source file, a makefile...
 14    files: "@SRCPATH@/*.c"
 15    # ===== FOR ANY BUILD SYSTEM
 16    # args: arguments forwarded to the build system
 17    # envs: environment variables exported to the target shell.
 18    # Only one build system is expected to be used at once among the following:
 19
 20    autotools:
 21      args: ['--disable-bootstrap']
 22      envs: []
 23    cmake:
 24      # "vars" is a shortcut to "args: ["-D<var>"]"
 25      vars: ['CMAKE_VERBOSE_MAKEFILE=ON']
 26      args: []
 27      envs: []
 28    make:
 29      # default makefile target to invoke
 30      target: "all"
 31      # a list of macros can be used as placeholder to be replaced when
 32      # unrolling. A list is available at the end of this file.
 33      args: ['MPICC=@COMPILER_CC@']
 34      envs: []
 35    sources:
 36      # program binary (if needed)
 37      binary: "a.out"
 38      # extra cflags
 39      cflags: "extra cflags"
 40      # extra ldflags
 41      ldflags: "extra ldflags"
 42      envs: []
 43      # force explicit language if PCVS has trouble to autodetect
 44      lang: ['cc', 'cxx', 'fc']
 45    custom:
 46      # Really complex build scenario PCVS may not be able to handle
 47      program: "myscript.sh"
 48      envs: ["VAR1=value", "VAR2=value"]
 49    # dependency scheme. This can be:
 50    # a fully expanded test name
 51    # a TE node (like ".this_is_a_job" for the current node).
 52    # A circular detection is run.
 53    depends_on: ["this_is_another_test"]
 54    # directory where program should be built
 55    cwd: "dir/to/build"
 56    # variants required from profiles to run this job
 57    # IF these variants are not satisfied, jobs will be discarded.
 58    # it indicates not support for a specific variant.
 59    # There is no preset variants, all of them can be declared by the profile.
 60    # It can be found under compiler.$LANG.variants
 61    variants:
 62      - openmp
 63      - accel
 64      - mpi
 65    # PM interaction, allow loading package/module BEFORE running jobs
 66    # This load will be done for every job, it may create overhead.
 67    # currently, only spack recipes & module will be allowed.
 68    # If a spack recipe is not available, it will be installed first.
 69    package_manager:
 70      spack:
 71        - protobuf@3.1.1
 72        - gcc@7.3.0
 73      module:
 74        - protobuf
 75        - gnu/gcc/7.3.0
 76
 77  run: &run_part
 78    # path to program (defaults to build.sources.binary if not set)
 79    program: "./a.out"
 80    # Scale the benchmark to build test scenarios.
 81    iterate:
 82      # subnodes are parsed relatively to the profile configuration.
 83      # There are expected to be declared to profiles, under "criterion"
 84      # only intersected values with profiles will be kept.
 85      n_mpi:
 86        values: [2, 4]
 87      n_omp:
 88        values: [1, 2]
 89      # Special key to declare program-scope parameter matrix.
 90      # One can use as many parameters to run the given program.
 91      # criterion name will be given by the key
 92      # criterion values will be given by the "values" field
 93      # criterion label (used to generate test-name) will be given by "subtitle"
 94      # parameter position will be given by "type": "argument" or "environment"
 95      # numeric attributes allow using sequence to generate values.
 96      # for instance, to generate only power of 2 values:
 97      #  => values: [{from: 0; to: 100, op: "mul": of: 2}]
 98      program:
 99        give_it_a_name:
100          numeric: true
101          type: "argument"
102          values: ["-iter 1000", "-fast"]
103          subtitle: "lol"
104    # directory where program should be built
105    cwd: "dir/to/build"
106    # dependency scheme, same rules as build.depends_on.
107    depends_on: ["this_is_another_run_test_in_the_same_file"]
108    # PM interaction, same as build.package_manager
109    package_manager:
110      spack:
111        - protobuf@3.1.1
112        - gcc@7.3.0
113      module:
114        - protobuf
115        - gnu/gcc/7.3.0
116  # How to validate jobs generated from this specification ?
117  # each of these subkeys are cumulative, all of them has to be true to 
118  # mark the job as a success
119  validate:
120    # expected return code
121    expect_exit: 0
122    # time limit
123    time:
124      # max execution time the job should not exceed to succeed
125      mean: 10.0
126      # time measurement may vary. Some tolerance is allowed (only upper-bound)
127      tolerance: 2.0
128      # After this time, job will be automatically stopped and marked as failed
129      # note that PCVS cannot be able to handle every job kill mechanism.
130      # Our approach is to terminate the process which spawned the test on
131      # resources. With some batch-managers, it may lead to leaks.
132      hard_timeout: 40
133      # Mark the job as failed without killing it.
134      # This allow you to run the test for code correctness on a slow machine
135      # without testing for speed.
136      soft_timeout: 20
137    # Output formatting compliance
138    # It is based on rule definition as regex. These regexs has to be matched (or
139    # not) to pass.
140    match:
141      # rule name
142      label:
143        # the regex
144        expr: '^\d+(\.\d+) received$'
145        # Is the pattern allowed or forbidden in the output.
146        # True: The regex MUST match to success
147        # False: The regex MUST NOT match to success
148        # default: True
149        expect: true|false
150      label2: 'Total Elapsed: \d+\.\d+ sec.$'
151    # Analysis method
152    # based on dedicated Python module (builtin or user-made) to be run on every
153    # jobs to select the final status
154    # use-case: mark job status depending on its execution history over past
155    # runs.
156    analysis:
157      method: "<method>"
158    # Generic scenario, not handled by PCVS.
159    # Job info & output are provided to the script, returning 0 in case of
160    # success, non-zero otherwise.
161    script:
162      path: "/path/to/script"
163
164  # Post execution rules.
165  # artifacts can be stored to the final results to store job-specific data.
166  # Currently, artifacts only supports filesystem
167  artifact:
168    # relative to $BUILDPATH
169    obj1: "./path/1"
170    obj2: "./path/2"
171
172  # From the output, extract data to be stored as results.
173  # A regex is used to capture the metric from the standard output and is stored
174  # to the final result JSON file for later reference.
175  metrics:
176    # metric name
177    metric1:
178      # regex to match
179      key: "regex"
180    metric2:
181      key: "regex"
182      attributes:
183        # if multiple occurences of the regex are found, only keep the first
184        # match. (default: false)
185        unique: true
186
187  # A list of properties to enable specific features.
188  # These attributes are common to build/run configurations
189  # if a different set of attributes are required for build & run a single TE,
190  # feel free to create TWO nodes, with a dep on "build".
191  attributes:
192    # control if the current benchmark program should be prefixed by
193    # the runtime.program command (default: true)
194    # True: ./a.out -> mpirun ./a.out
195    # False: ./a.out -> ./a.out
196    command_wrap: true
197    # controls if the relative path defined by this TE should be converted to
198    # absolute path to avoid PATH resolution. (default: true)
199    # True: ./a.out -> @BUILDIR@/./a.out
200    # False: spack install -> spack install
201    path_resolution: true
202    # duplicates inputs defined by this TE for any generated test.
203    # For instance, if the current build directory provide a non-shareable
204    # resource, it may run into trouble when running concurrent tests from this
205    # specification. A dedicated directory is built for each subtest.
206    # (default: false)
207    copy_input: false
208    # duplicates outputs defined by this TE for any generated test.
209    # This is the opposite of copy_input, to allow generating concurrent artifacts
210    # from this single specification (default: false)
211    copy_output: false
